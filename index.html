function doGet(e) { var type = (e && e.parameter && e.parameter.type) ? e.parameter.type : "stock"; var ss = SpreadsheetApp.getActiveSpreadsheet(); try { if (type === "sector") { var sectorSheet = ss.getSheetByName("업종시세_VIEW"); // 시트명 정확히 일치해야 함 // [상단 통계] C5:F5 데이터 추출 (image_d32560.png 기준) var statsData = sectorSheet.getRange("C5:F5").getValues()[0]; var statsText = "전체 " + statsData[0] + " 상승 " + statsData[1] + " 보합 " + statsData[2] + " 하락 " + statsData[3]; // [업종 리스트] 상위(A5:B9) + 하위(A13:B17) 통합 var top = sectorSheet.getRange("A5:B9").getValues(); var bottom = sectorSheet.getRange("A13:B17").getValues(); var allSectors = top.concat(bottom); var dataResults = []; for (var i = 0; i < allSectors.length; i++) { if (allSectors[i][0]) { var rateVal = parseFloat(allSectors[i][1]); dataResults.push({ "name": String(allSectors[i][0]), "rate": isNaN(rateVal) ? "0.00%" : (rateVal * 100).toFixed(2) + "%", "status": rateVal > 0 ? "상승" : (rateVal < 0 ? "하락" : "보합") }); } } return createJsonResponse({"stats": statsText, "data": dataResults}); } else { var viewSheet = ss.getSheetByName("KVA_VIEW"); var stockData = viewSheet.getRange("A3:F32").getValues(); var dataResults = []; for (var j = 0; j < stockData.length; j++) { if (stockData[j][1]) { var sRateVal = parseFloat(stockData[j][3]); dataResults.push({ "rank": stockData[j][0], "name": stockData[j][1], "price": Number(stockData[j][2]).toLocaleString(), "rate": isNaN(sRateVal) ? "0.00%" : (sRateVal * 100).toFixed(2) + "%", "change": (stockData[j][5] > 0 ? "+" : "") + Number(stockData[j][5]).toLocaleString(), // 등락액 복구 "volume": formatVolume(stockData[j][4]), "status": sRateVal > 0 ? "상승" : (sRateVal < 0 ? "하락" : "보합") }); } } return createJsonResponse({"data": dataResults}); } } catch (err) { return createJsonResponse({"error": err.toString()}); } } // 웹 브라우저 연결을 확실하게 해주는 응답 함수 function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function formatVolume(num) { var v = Number(num); if (isNaN(v)) return "0"; if (v >= 1000000) return (v / 1000000).toFixed(1) + "M"; if (v >= 1000) return (v / 1000).toFixed(1) + "K"; return v.toString(); }
